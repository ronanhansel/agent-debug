   Example base R replacement for summarySE:
   summarySE_base <- function(data, measurevar, groupvars) {
     result <- aggregate(data[[measurevar]], by=data[groupvars], FUN=function(x) {
       c(mean=mean(x, na.rm=TRUE), sd=sd(x, na.rm=TRUE), n=length(x))
     })
     result$se <- result$x[,"sd"] / sqrt(result$x[,"n"])
     result
   }
